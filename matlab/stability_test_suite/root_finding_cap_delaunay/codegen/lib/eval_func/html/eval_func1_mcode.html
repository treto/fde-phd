1% @brief: Verifies stability of a discrete-time, integer-order system by checking i
 2% eigenvalues of the characteristic polynomial all lay inside the complex
 3% unit circle
 4% 
 5% @param transfer_fun TransferFunction objject
 6%
 7% @retval boolean value, true if system is stable, false otherwise
 8function point = eval_func(Gz, Gp, input)
 9    output = 1;
10    for i=1:numel(Gz)
11        output = output*(input - Gz(i));
12    end
13
14    for i=1:numel(Gp)
15        output = output/(input - Gp(i));
16    end
17    point = output;
18end
